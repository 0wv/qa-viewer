# 完全一致問題形式
#
# 例１）
q1:=a1

# "q1" の答えは "a1" です。
#
# ":=" を使用して OR 条件で区切れます。
# 例２）
q2:=a1:=a2

# "q2" の答えは "a1" または "a2" です。

# 完全一致選択問題形式
#
# 例３）
q3:-a1:-a2:-a3:=1

# 選択可能な答えを ":-" で区切ります。
# "q3" の答えは "a1" | "a2" | "a3" の中から選びます。
# ":=" の後に答えが何番目か表記します。
# 例３では１番目 ("a1") です。
#
# 完全一致選択問題形式では ":=" は AND 条件を表します。
#
# 例４）
q4:-a1:-a2:-a3:=1:=2

# "q4" の答えは "a1" と "a2" です。
